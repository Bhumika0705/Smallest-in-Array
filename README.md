# Smallest-in-Array
Smallest element in Array
public class SmallinArray {
    public static void main(String[] args) {
    int arr[] = {2,3,5,7,8,9};
    int max = arr[0];
    for (int i = 0;i< arr.length;i++){
        if (max<arr[i]){
            max=arr[i];
        }
    System.out.println(arr[i]);
    }
    }
    }
